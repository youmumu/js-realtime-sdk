decorator，定义消息类的自定义字段
Example
@messageField(['foo'])
class CustomMessage extends TypedMessage {
  constructor(foo) {
    super();
    this.foo = foo;
  }
}

// 不支持 decorator 的情况下可以这样使用
class CustomMessage extends TypedMessage {
  constructor(foo) {
    super();
    this.foo = foo;
  }
  //...
}
messageField(['foo'])(CustomMessage);
